library(dplyr)
library(ggplot2)
library(survival)

data(accel)

t1 = accel$time[50000]

accel |>
  filter(time > t1 & time <= t1 + minutes(5)) |>
  spec_sig(take_log = TRUE) |>
  filter(freq <= 3) |>
  plot_accel(x_var = "freq") +
    theme_bw()
